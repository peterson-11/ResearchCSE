def data_cleaner(criterion):
  if criterion[0] != 'Both':
    cleaned = mental_df.loc[mental_df['Indicator'] == criterion[0]]
  cleaned  = cleaned.loc[mental_df['Group'] == criterion[1]]
  if criterion[2] != 'x':
    cleaned  = cleaned.loc[mental_df['Subgroup'] == criterion[2]]
  if criterion[3] != 'x':
    cleaned  = cleaned.loc[mental_df['State'] == criterion[3]]
  return cleaned

### Function to create criteria for selection

def criteria():
  crit_list = []
  illness = input('Anxiety or Depression?')
  if illness == 'Anxiety':
    crit_list.append('Symptoms of Anxiety Disorder')
  elif illness == 'Depression':
    crit_list.append('Symptoms of Depressive Disorder')
  else:
    crit_list.append(illness)
  group = input('What is the primary comparison?')
  crit_list.append(group)
  sub = input('What is the secondary level you would like to compare at? Press x for all subgroups')
  crit_list.append(sub)
  level = input('What location scope are you looking at? Press x for all location levels')
  crit_list.append(level)
  return crit_list
